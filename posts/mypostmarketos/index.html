安装后的初步配置#

Here is a script to set dynamic wallpaper in KDE

#!/usr/bin/env sh

sleep 2
TEMP=0
MINS=10
# Path of Wallpaper
location="/home/johnny/.config/wallpapers/"
wallpaper=`date +%H`.jpg

# Set Origin Wallpaper
plasma-apply-wallpaperimage $location$wallpaper

while true;do

    let TEMP++

    if [ $TEMP -eq $MINS ];
    then
        TEMP=0
        wallpaper=`date +%H`.jpg
        
        plasma-apply-wallpaperimage $location$wallpaper
    fi

    sleep 60
done &
